Part 3: Google Apps Script (GAS) 自動化

自動更新與備份腳本

程式碼

功能：每天自動將 IMPORTFEED 抓到的動態資料，備份到新的工作表中存檔。

function myFunction() {
    // 1. 指定URL開啟試算表 (請輸入自己Google Spreadsheet文件的實際URL路徑)
    var url = "https://docs.google.com/spreadsheets/d/1_TZRxwYQHQvB2zWHOohT0pGh7UVlvKrAOpbRK56601w/edit";   
    var spredSheet = SpreadsheetApp.openByUrl(url);

    // 2. 取得日期 (以取得時間命名)
    var d = new Date();
    var targetName = Utilities.formatDate(d, "GMT+8", "yyyy-MM-dd HH:mm:ss");

    // 3. 開啟工作表和目標工作表
    var baseSheet = spredSheet.getSheetByName("original");
    var targetSheet = createSheet(spredSheet, targetName, 0);

    // 4. 複製取得值至目標工作表
    // 指定複製範圍（A2～d101）
    var rangeToCopy = baseSheet.getRange('A2:D101');
    
    //指定複製目的地的起始儲存格（A2）
    var targetToCopy = targetSheet.getRange('A2:D101');
    rangeToCopy.copyTo(targetToCopy);
    targetSheet.getRange(1,1).setValue("Title");
    targetSheet.getRange(1,2).setValue("Url");
    targetSheet.getRange(1,3).setValue("Date Created");
    targetSheet.getRange(1,4).setValue("Summary");
    rangeToCopy.copyTo(targetToCopy);

    // 5. 再次使用ImportFeed函數
    baseSheet.getRange(1,1).setValue("");
    baseSheet.getRange(1,1).setFormula("IMPORTfeed(\"https://rss.itunes.apple.com/api/v1/tw/music-videos/top-music-videos/all/100/explicit.rss\",\"items\",true,100)");  
    // \反斜線為跳字元 : 放置在雙引號前，使雙引號功能無效    
}

function createSheet(spredSheet, sheetName, index) {
    var sheet = spredSheet.getSheetByName(sheetName);
    if ( sheet == null) {
        spredSheet.insertSheet(sheetName, index);
        sheet = spredSheet.getSheetByName(sheetName);
    }
    return sheet;
}

Part 4: VBA 網頁爬蟲 (補充教材)

注意：以下範例使用 InternetExplorer 物件，適用於舊版環境或特定需求。

1. 基礎爬蟲：三民書局排行

VBA 
Option Explicit
#If Win64 Then
  Declare PtrSafe Sub Sleep Lib "kernel32" (ByVal dwMilliseconds As Long)                        '(3)
#Else
    Declare Sub Sleep Lib "kernel32" (ByVal dwMilliseconds As Long)                               '(4)
#End If


Private objIE As InternetExplorer

Sub sanmin()
    Dim url, row   '宣告變數
    On Error Resume Next
    row = 1

        url = "https://www.sanmin.com.tw/promote/top"  '三民書局
    
    Set objIE = New InternetExplorer  '建立IE物件
    
        objIE.Visible = False
        objIE.Navigate2 (url)


'等待讀取完成

        While objIE.readyState <> READYSTATE_COMPLETE Or objIE.Busy = True
            DoEvents
            Sleep 100
        Wend
        Sleep 100

        Dim objDoc As HTMLElementCollection
        Set objDoc = objIE.document

        Dim BookName As IHTMLElement   '宣告變數指定型態
 

        For Each BookName In objDoc.getElementsByClassName("Title") '確認書名在網頁的位置
            Worksheets(1).Cells(row, 1) = row                       '寫入工作表與儲存格
            Worksheets(1).Cells(row, 2) = BookName.innerText        'innerText 取得標籤間的文字
            row = row + 1
        Next
    
    Set objIE = Nothing
    MsgBox "Done ！"
End Sub

3. Children 節點遍歷與翻頁

VBA 
Option Explicit
#If Win64 Then
  Declare PtrSafe Sub Sleep Lib "kernel32" (ByVal dwMilliseconds As Long)                        '(3)
#Else
    Declare Sub Sleep Lib "kernel32" (ByVal dwMilliseconds As Long)                               '(4)
#End If


Private objIE As InternetExplorer

Sub sanmin4()
    
    Dim url, row, n   '宣告變數
    On Error Resume Next
    row = 1
    For n = 1 To 3
        url = "https://www.sanmin.com.tw/promote/top/?id=wbyy&pi=" & n '三民書局
    
    Set objIE = New InternetExplorer     '建立IE物件
    
        objIE.Visible = False
        objIE.Navigate2 (url)


'等待讀取完成

        While objIE.readyState <> READYSTATE_COMPLETE Or objIE.Busy = True
            DoEvents
            Sleep 100
        Wend
        Sleep 100

        Dim objDoc As HTMLElementCollection
        Set objDoc = objIE.document
        Dim BookName As IHTMLElement   '宣告變數指定型態
 

        For Each BookName In objDoc.getElementsByClassName("Info") '確認書名在網頁的位置
            Worksheets(4).Cells(row, 1) = row                                           '寫入儲存格
            Worksheets(4).Cells(row, 2) = BookName.Children(0).Children(0).innerText    'BookName.Children(0).innerText 執行結果相同
            Worksheets(4).Cells(row, 3) = BookName.Children(1).Children(0).innerText
            Worksheets(4).Cells(row, 4) = BookName.Children(1).Children(1).innerText
            Worksheets(4).Cells(row, 5) = BookName.Children(2).Children(0).Children(0).innerText
            Worksheets(4).Cells(row, 6) = BookName.Children(2).Children(0).Children(1).innerText
            
            row = row + 1
        Next
        
    Next n
    
    objIE.Quit
    Set objIE = Nothing
    MsgBox "Done ！"
End Sub
